38 OSS_RESULTCODE OSSMultipartOperation::UploadPart(UploadPartRequest & uploadPartRequest,UploadPartResult & result)
39 {
40  map<string,string> headers;
41  map<string,string> params;
42  map<string,string> userInfo;
43  userInfo.insert(map<string, string> :: value_type("method","UploadPart"));
44  if(uploadPartRequest.bucketName.empty() || uploadPartRequest.key.empty()
45  ||uploadPartRequest.uploadId.empty() )
46  {
47  return OSS_RESULTCODE_INVALID;
48  }
49  int l = uploadPartRequest.partSize;
50  int i = uploadPartRequest.partNumber;
51  if((l < 0) || (l > 5368709120) ||((i < 0) || (i > 10000)))
52  {
53  return OSS_RESULTCODE_INVALID;
54  }
55  char temp[64];
56  sprintf(temp,"%ld",l);
57  string strLength(temp);
58  headers.insert(map<string, string> :: value_type("Content-Length",strLength));
59  headers.insert(map<string, string> :: value_type("Content-MD5",uploadPartRequest.md5Digest));
60  char temp1[64];
61  sprintf(temp1,"%d",uploadPartRequest.partNumber);
62  string strPartNumber(temp1);
63  params.insert(map<string, string> :: value_type("partNumber",strPartNumber));
64  params.insert(map<string, string> :: value_type("uploadId",uploadPartRequest.uploadId));
65 
66  ResponseMessage rm ;
67  rm = invoke(HttpMethod_PUT,uploadPartRequest.bucketName,uploadPartRequest.key,headers,params,uploadPartRequest.data,l,userInfo);
68  OSS_RESULTCODE rCode;
69  if(rm.isSuccessful())
70  {
71  rCode = OSS_RESULTCODE_OK;
72  map<string,string> headers =rm.getHeaders();
73  map<string,string>::iterator iter = headers.find("ETag");
74  string eTag="";
75  if(iter != headers.end())
76  {
77  eTag = iter->second;
78  }
79  eTag = StringUtil::trimQuote(eTag);
80  result.partNumber = i;
81  result.eTag = eTag;
82 
83  }
84  else
85  {
86  string strError(rm.getContent());
87  OSSError error(strError);
88  rCode = OSSUtil::codeFromString(error.code);
89  }
90  return rCode;

503 void OSSMultipartOperation::serviceClientRequestFinished(DefaultServiceClient*defaultServiceClient,ResponseMessage &responseMessage)
504 {
505  if(delegate != NULL)
506  {
507  map<string,string> userInfo = responseMessage.userInfo;
508  map<string,string>::iterator iter ;
509  iter = userInfo.find("method");
510  string method = "";
511  if(iter != userInfo.end())
512  {
513  method = iter->second;
514  }
515  if (responseMessage.isSuccessful())
516  {
517  if(method.compare("AbortMultipartUpload")==0)
518  {
519  string uploadId = userInfo["uploadId"];
520  delegate->OnAbortMultipartUploadComplete(this,uploadId);
521  }
522  else if(method.compare("InitiateMultipartUpload")==0)
523  {
524  string strXML="";
525  InitiateMultipartUploadResult result;
526  if(responseMessage.getContent() != NULL)
527  {
528  strXML.append(responseMessage.getContent());
529  result.MakeInitiateMultipartUploadResult(strXML);
530  }
531  delegate->OnInitiateMultipartUploadComplete(this,result);
532  }
533  else if(method.compare("CompleteMultipartUpload")==0)
534  {
535  string strXML ;
536  CompleteMultipartUploadResult result;
537  if(responseMessage.getContent() != NULL)
538  {
539  strXML.append(responseMessage.getContent());
540  result.MakeCompleteMultipartUploadResult(strXML);
541  }
542  delegate->OnCompleteMultipartUploadComplete(this,result);
543  }
544  //ListParts
545  else if(method.compare("ListParts")==0)
546  {
547  string strXML ;
548  PartListing result;
549  if(responseMessage.getContent() != NULL)
550  {
551  strXML.append(responseMessage.getContent());
552  result.MakePartListing(strXML);
553  }
554  delegate->OnListPartsComplete(this,result);
555  }
556  //ListMultipartUploads
557  else if(method.compare("ListMultipartUploads")==0)
558  {
559  string strXML ;
560  MultipartUploadListing result;
561  if(responseMessage.getContent() != NULL)
562  {
563  strXML.append(responseMessage.getContent());
