39 OSS_RESULTCODE OSSObjectGroupOperation::PostObjectGroup(PostObjectGroupRequest &postObjectGroupRequest,PostObjectGroupResult & result)
40 {
41  map<string,string> headers;
42  map<string,string> params;
43  map<string,string> userInfo;
44  userInfo.insert(map<string, string> :: value_type("method","PostObjectGroup"));
45 
46  if(postObjectGroupRequest.bucketName.empty() ||postObjectGroupRequest.objectGroupName.empty()
47  ||postObjectGroupRequest.objectGroupPartETags.size() <= 0 || postObjectGroupRequest.objectGroupPartETags.size() >1000)
48  {
49  return OSS_RESULTCODE_INVALID;
50  }
51  std::sort(postObjectGroupRequest.objectGroupPartETags.begin(), postObjectGroupRequest.objectGroupPartETags.end(),ObjectGroupPartETag::cmp);
52  params.insert(map<string, string> :: value_type("group",""));
53  string strXMLContent = MakePostObjectGroupXML(postObjectGroupRequest.objectGroupPartETags);
54 
55  string utf8XML =HttpUtil::MbcsToUtf8(strXMLContent);
56 
57  headers.insert(map<string, string> :: value_type("Content-Type","text/plain"));
58  char * data =(char*) utf8XML.c_str();
59  ResponseMessage rm = invoke(HttpMethod_POST,postObjectGroupRequest.bucketName,postObjectGroupRequest.objectGroupName,headers,params,data,utf8XML.size(),userInfo);
60  OSS_RESULTCODE rCode;
61  if(rm.isSuccessful())
62  {
63  rCode = OSS_RESULTCODE_OK;
64  string strXML ;
65  if(rm.getContent() != NULL)
66  {
67  strXML.append(rm.getContent());
68  result.MakePostObjectGroupResult(strXML);
69  }
70  }
71  else
72  {
73  string strError(rm.getContent());
74  OSSError error(strError);
75  rCode = OSSUtil::codeFromString(error.code);
76  }
77  return rCode;

216 void OSSObjectGroupOperation::serviceClientRequestFinished(DefaultServiceClient*defaultServiceClient,ResponseMessage &responseMessage)
217 {
218  if(delegate != NULL)
219  {
220  map<string,string> userInfo = responseMessage.userInfo;
221  map<string,string>::iterator iter ;
222  iter = userInfo.find("method");
223  string method = "";
224  if(iter != userInfo.end())
225  {
226  method = iter->second;
227  }
228  if (responseMessage.isSuccessful())
229  {
230  if(method.compare("PostObjectGroup")==0)
231  {
232  string strXML ;
233  PostObjectGroupResult result;
234  if(responseMessage.getContent() != NULL)
235  {
236  strXML.append(responseMessage.getContent());
237  result.MakePostObjectGroupResult(strXML);
238  }
239  delegate->OnPostObjectGroupComplete(this,result);
240  }
241  else if(method.compare("GetObjectGroupIndex")==0)
242  {
243  string strXML ;
244  FetchObjectGroupIndexResult result;
245  if(responseMessage.getContent() != NULL)
246  {
247  strXML.append(responseMessage.getContent());
248  result.MakeFetchObjectGroupIndexResult(strXML);
249  delegate->OnGetObjectGroupIndexComplete(this,result);
250  }
251  }
252 
253  }
254  else
255  {
256  OSS_RESULTCODE rCode;
257  string strError(responseMessage.getContent());
258  OSSError error(strError);
259  rCode = OSSUtil::codeFromString(error.code);
260  //PostObjectGroup
261  if(method.compare("PostObjectGroup")==0)
262  {
263  delegate->OnPostObjectGroupFailed(this,rCode);
264  }
265  //GetObjectGroupIndex
266  else if(method.compare("GetObjectGroupIndex")==0)
267  {
268  delegate->OnGetObjectGroupIndexFailed(this,rCode);
269  }
270 
271  }

283 string OSSObjectGroupOperation::MakePostObjectGroupXML(vector<ObjectGroupPartETag> &objectGroupPartETags)
284 {
285  string rtnString="";
286  rtnString+="<CreateFileGroup>";
287  for(int i=0;i<objectGroupPartETags.size();i++)
288  {
289  ObjectGroupPartETag partEtag = objectGroupPartETags[i];
290  rtnString+="<Part>";
291  rtnString+="<PartNumber>";
292  char cPartNumber[64];
293  sprintf(cPartNumber,"%ld",partEtag.partNumber);
294  rtnString+=cPartNumber;
295  rtnString+="</PartNumber>";
296  rtnString+="<PartName>";
297  rtnString+=partEtag.partName;
298  rtnString+="</PartName>";
299  rtnString+="<ETag>";
300  rtnString+=partEtag.eTag;
301  rtnString+="</ETag>";
302  rtnString+="</Part>";
303  }
304 
305  rtnString+="</CreateFileGroup>";
306  return rtnString;
