103 void OSSUtil::populateRequestMetadata(map<string,string> &header,ObjectMetadata &objectMetadata)
104 {
105  bool isHaveContentType = false;
106  map<string,string> metadata = objectMetadata.GetMetadata();
107  map<string,string> userMetadata = objectMetadata.GetUserMetadata();
108  map<string,string>::iterator iter;
109  if(metadata.size() > 0)

111  for(iter = metadata.begin();iter != metadata.end();iter++)
112  {
113  string strKey = iter->first;
114  if(strKey.compare("Content-Type") ==0)
115  {
116  isHaveContentType = true;

118  header.insert(map<string, string> :: value_type(iter->first,iter->second));
119  }
120  }
121  if(!isHaveContentType)
122  {
123  header.insert(map<string, string> :: value_type("Content-Type","application/octect-stream"));
124  }
125  if(userMetadata.size() > 0)
126  {
127  for(iter = userMetadata.begin();iter != userMetadata.end();iter++)
128  {
129  string strKey = iter->first;
130  string strValue = iter->second;
