304 conv_num(const char **buf, int *dest, int llim, int ulim)
305 {
306  int result = 0;
307 
308  /* The limit also determines the number of valid digits. */
309  int rulim = ulim;
310 
311  if (**buf < '0' || **buf > '9')
312  return (0);
313 
314  do {
315  result *= 10;
316  result += *(*buf)++ - '0';
317  rulim /= 10;
318  } while ((result * 10 <= ulim) && rulim && **buf >= '0' && **buf <= '9');
319 
320  if (result < llim || result > ulim)
321  return (0);
322
