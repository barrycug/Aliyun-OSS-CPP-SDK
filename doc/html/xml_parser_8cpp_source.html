756 {

758  // sequences &amp;, &quot;, &apos;, &lt;, &gt; and replace them by the characters
759  // &,",',<,>. This function is used internally by the XML Parser. All the calls to
760  // the XML library will always gives you back "decoded" strings.

763  // out: new allocated string converted from xml
764  if (!s) return NULL;

770  while ((lo>0)&&(*s))

797  }
798  }
799  } else
